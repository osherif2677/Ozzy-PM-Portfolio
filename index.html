SQL-style ¬∑ INF / SDU aggregation

-- INF and SDU by department & month
SELECT
  DATE_TRUNC('month', order_date) AS month,
  department,
  SUM(CASE WHEN outcome = 'INF' THEN 1 ELSE 0 END)::float /
  COUNT(*)                         AS inf_pct,
  SUM(CASE WHEN outcome = 'SDU' THEN 1 ELSE 0 END)::float /
  COUNT(*)                         AS sdu_pct
FROM outbound_orders
GROUP BY 1, 2
ORDER BY 1, 2;
