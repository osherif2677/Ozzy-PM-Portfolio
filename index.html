Tools & snippets

Excel + SQL + Python

Simple examples of how I’d work with inventory data in different tools.

=SUMIFS([INF_Units],[Dept],"Paleo",[Month],$B$1)
/* Highlight Paleo SKUs driving INF in a given month */

=IF([@INF_pct]>0.10,"High","OK")
/* Flag items over 10% INF for coaching */
SELECT department,
       SUM(inf_units)   AS inf_units,
       SUM(order_units) AS order_units,
       SUM(inf_units) * 1.0 / NULLIF(SUM(order_units),0) AS inf_pct
FROM   outbound_orders
WHERE  site = 'MAG5'
  AND  order_date BETWEEN '2025-11-01' AND '2025-11-30'
GROUP BY department
ORDER BY inf_pct DESC;
import pandas as pd

df = pd.read_csv("inventory_health_mag5.csv")
pivot = df.pivot_table(
    index="quarter",
    values=["hp_agile","produce_culling","exp_compliance","shrink_pct"]
)
pivot.plot(kind="bar", title="MAG5 inventory execution")
